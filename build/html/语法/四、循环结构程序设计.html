求整数的和与均值

#include <bits/stdc++.h>

using namespace std;

int sum;
double ave;

int main()
{
	int n;
	cin >> n;

	for (int i = 0; i < n; i++)
	{
		int x;
		cin >> x;

		sum += x;
	}

	ave = sum * 1.0 / n;

	printf("%d %.5lf\n", sum, ave);

	return 0;
}

最高的分数

//介绍”打擂台“

#include <bits/stdc++.h>

using namespace std;

int ans = -1;

int main()
{
	int n;
	cin >> n;

	while (n--)
	{
		int x;
		cin >> x;

		ans = max(ans, x);
	}

	cout << ans << endl;

	return 0;
}
