3610 - 单向链表操作

```cpp
#include <iostream>
#include <cstdio>

using namespace std;

const int N = 5010;

int head, e[N], ne[N], idx;

void init()
{
	head = 0, idx = 1;
	
	e[0] = 0, ne[0] = -1;
}

bool find(int x)
{
	for (int i = head; i != -1; i = ne[i])
		if (e[i] == x) return true;

	return false;
}

void add(int x, int y)
{
	for (int i = head; i != -1; i = ne[i])
		if (e[i] == x)
			e[idx] = y, ne[idx] = ne[i], ne[i] = idx++;
}

void remove(int x)
{
	for (int i = head; i != -1; i = ne[i])
		if (e[ne[i]] == x)
			ne[i] = ne[ne[i]];
}

void show()
{
	for (int i = head; i != -1; i = ne[i])
		printf("%d ", e[i]);
	puts("");

}

int main()
{
	init();

	int n;
	scanf("%d", &n);

	int op, x, y;
	while (n--)
	{
		scanf("%d", &op);

		if (op == 1)
		{
			scanf("%d%d", &x, &y);

			if (!find(x) || find(y)) 
				printf("error\n");
			else
			{
				add(x, y);
				printf("ok\n");
			}
		}
		else
		{
			scanf("%d", &x);

			if (!find(x))
				printf("error\n");
			else
			{
				remove(x);
				printf("ok\n");
			}
		}
	}

	show();

	return 0;
}
```



3620 - 约瑟夫环

```cpp
#include <iostream>
#include <queue>

using namespace std;

queue<int> q;
int n, m;

int main()
{
	cin >> n >> m;

	for (int i = 0; i < n; i++) q.push(i);

	while (q.size() > 1)
	{
		for (int i = 1; i < m; i++)
		{
			int t = q.front();
			q.pop();

			q.push(t);
		}
		
		q.pop();
	}

	cout << q.front() << endl;

	return 0;
}
```

```cpp
#include <iostream>
#include <cstdio>

using namespace std;

const int N = 1e6 + 10;

int e[N], l[N], r[N], idx;
int n, m;

void init()
{
	for (int i = 0; i < n; i++)
	{
		e[i] = i;
		r[i] = (i + 1) % n;
		l[i] = (i - 1 + n) % n;
	}
}

int find(int k)
{
	//报m的出队，从x往后找m-1个人
	int t = k;
	for (int i = 1; i < m; i++)
		t = r[t];
	
	return t;
}

void remove(int k)
{
	r[l[k]] = r[k];
	l[r[k]] = l[k];
}

int main()
{
	scanf("%d%d", &n, &m);

	init();
	
	int k = 0;
	for (int i = 1; i < n; i++)
	{
		k = find(k);
		remove(k);
		
		k = r[k];
	}

	printf("%d\n", k);

	return 0;
}
```

3630 - 大整数查找

```cpp
#include <iostream>
#include <cstdio>
#include <map>

using namespace std;

const int N = 20;

map<string, int> mp;
char num[N];
int n;

int main()
{
	scanf("%d", &n);

	while (n--)
	{
		scanf("%s", num);

		if (mp.find(num) == mp.end())
		{
			printf("accepted\n");
			mp[num]++;
		}
		else

	return 0;
```

```cpp
#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

const int N = 100010, P = 100003;

vector<int> h[N];
int n;

bool find(int x)
{
	int pos = x % P;
	for (int i = 0; i < h[pos].size(); i++)
		if (h[pos][i] == x) return false;

	return true;
}

void insert(int x)
{
	int pos = x % P;
	h[pos].push_back(x);
}

int main()
{
	scanf("%d", &n);

	while (n--)
	{
		int num;
		scanf("%d", &num);

		if (find(num))
		{
			insert(num);
			printf("accepted\n");
		}
		else
			printf("ignored\n");
	}

	return 0;
}
```
