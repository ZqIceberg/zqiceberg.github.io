#### T133420 走迷宫

```cpp
#include <iostream>
#include <cstdio>
#include <queue>
#include <cstring>

using namespace std;

typedef pair<int, int> PII;

const int N = 110;

int g[N][N], d[N][N];
int n, m;

int bfs()
{
	memset(d, -1, sizeof d);
	
	queue<PII> q;
	d[0][0] = 0;
	q.push(PII(0, 0));

	int dx[] = {0, 1, 0, -1}, dy[] = {1, 0, -1, 0};
	while (q.size() != 0)
	{
		PII t = q.front();
		q.pop();

		for (int i = 0; i < 4; i++)
		{
			int x = t.first + dx[i], y = t.second + dy[i];
			if (x < 0 || x >= n || y < 0 || y >= m) continue;
			if (g[x][y] == 1 || d[x][y] != -1) continue;

			d[x][y] = d[t.first][t.second] + 1;
			q.push(PII(x, y));

			/*
			if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == 0 && d[x][y] == -1)
			{
				d[x][y] = d[t.first][t.second] + 1;
				q.push(PII(x, y));
			}
			*/
		}
	}

	return d[n - 1][m - 1];
}

int main()
{
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> g[i][j];

	cout << bfs() << endl;

	return 0;
}
```

#### P1451 求细胞数量
flood fill
```cpp
#include <iostream>
#include <queue>
#include <cstdio>

using namespace std;

typedef pair<int, int> PII;

const int N = 110;

char g[N][N];
int n, m;
int res;

void bfs(int x, int y)
{
	queue<PII> q;
	q.push(PII(x, y));

	while (q.size())
	{
		PII t = q.front();
		q.pop();

		int dx[] = {0, 1, 0, -1}, dy[] = {1, 0, -1, 0};
		for (int i = 0; i < 4; i++)
		{
			int x = t.first + dx[i], y = t.second + dy[i];
			
			if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] != '0')
			{
				q.push(PII(x, y));
				g[x][y] = '0';
			}
		}
	}

	res++;
}

int main()
{
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> g[i][j];

	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			if (g[i][j] != '0')
				bfs(i, j);

	cout << res << endl;

	return 0;
}

```

#### P1443 马的遍历【扩展】

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
#include <fstream>

#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;

const int N = 410, M = N * N;

PII q[M];
bool st[N][N];
int dist[N][N];

int sx, sy;
int n, m;

int dx[] = {-2, -1, 1, 2, 2, 1, -1, -2};
int dy[] = {1, 2, 2, 1, -1, -2, -2, -1};

void bfs()
{   
    int hh = 0, tt = 0; 
    q[0].x = sx, q[0].y = sy;
    dist[sx][sy] = 0;
    st[sx][sy] = true;
    
    while (hh <= tt)
    {   
        PII t = q[hh++];
        //printf("---%d %d   %d\n", t.x, t.y, g[t.x][t.y].x);
        
        for (int i = 0; i < 8; i++)
        {
            int a = t.x + dx[i], b = t.y + dy[i];
            if (a <= 0 || a > n || b <= 0 || b > m) continue;
            if (st[a][b]) continue;

            ++tt;
            q[tt].x = a, q[tt].y = b;
            dist[a][b] = dist[t.x][t.y] + 1;
            st[a][b] = true;
        }
    }
}

int main()
{
    //freopen("horse.in", "r", stdin);

    cin >> n >> m >> sx >> sy;

    memset(dist, -1, sizeof dist);

    bfs();

    for(int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= m; j++)
            printf("%-5d", dist[i][j]);
        puts("");
    }
}
```
