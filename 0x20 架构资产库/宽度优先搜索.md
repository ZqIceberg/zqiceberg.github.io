#### 走迷宫

```cpp
#include <iostream>
#include <cstdio>
#include <queue>
#include <cstring>

using namespace std;

typedef pair<int, int> PII;

const int N = 110;

int g[N][N], d[N][N];
int n, m;

int bfs()
{
	memset(d, -1, sizeof d);
	
	queue<PII> q;
	d[0][0] = 0;
	q.push(PII(0, 0));

	int dx[] = {0, 1, 0, -1}, dy[] = {1, 0, -1, 0};
	while (q.size() != 0)
	{
		PII t = q.front();
		q.pop();

		for (int i = 0; i < 4; i++)
		{
			int x = t.first + dx[i], y = t.second + dy[i];
			if (x < 0 || x >= n || y < 0 || y >= m) continue;
			if (g[x][y] == 1 || d[x][y] != -1) continue;

			d[x][y] = d[t.first][t.second] + 1;
			q.push(PII(x, y));

			/*
			if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == 0 && d[x][y] == -1)
			{
				d[x][y] = d[t.first][t.second] + 1;
				q.push(PII(x, y));
			}
			*/
		}
	}

	return d[n - 1][m - 1];
}

int main()
{
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> g[i][j];

	cout << bfs() << endl;

	return 0;
}
```

#### P1451 求细胞数量
flood fill
```cpp
#include <iostream>
#include <queue>
#include <cstdio>

using namespace std;

typedef pair<int, int> PII;

const int N = 110;

char g[N][N];
int n, m;
int res;

void bfs(int x, int y)
{
	queue<PII> q;
	q.push(PII(x, y));

	while (q.size())
	{
		PII t = q.front();
		q.pop();

		int dx[] = {0, 1, 0, -1}, dy[] = {1, 0, -1, 0};
		for (int i = 0; i < 4; i++)
		{
			int x = t.first + dx[i], y = t.second + dy[i];
			
			if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] != '0')
			{
				q.push(PII(x, y));
				g[x][y] = '0';
			}
		}
	}

	res++;
}

int main()
{
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> g[i][j];

	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			if (g[i][j] != '0')
				bfs(i, j);

	cout << res << endl;

	return 0;
}

```

#### P1996 约瑟夫问题
```cpp
#include <iostream>
#include <queue>

using namespace std;

queue<int> q;
int n, m;

int main()
{
	cin >> n >> m;

	for (int i = 1; i <= n; i++) q.push(i);

	while (q.size())
	{
		for (int i = 1; i < m; i++)
		{
			int t = q.front();
			q.pop();

			q.push(t);
		}

		cout << q.front() << ' ';
		q.pop();
	}
	cout << endl;

	return 0;
}
```
